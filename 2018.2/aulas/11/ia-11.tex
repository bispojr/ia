	\begin{frame}{Questão 017}	
		\begin{block}{[Q017]}
			Sobre os algoritmos de busca local, é \underline{incorreto} afirmar que...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item são bastante utilizados em aplicações comerciais e industriais.
			\item encontram soluções razoáveis em grandes espaços de estados.
			\item usam pouquíssima memória, quase sempre um valor constante.
			\item costumam guardar na memória o caminho que leva até à solução do problema.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 020}	
		\begin{block}{[Q020]}
			A busca da têmpera simulada difere da busca de subida de encosta em relação...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item ao cálculo da função objetivo e da função heurística.
			\item em relação aos critérios de desempenho, pois a busca gulosa local é ótima e completa.
			\item à possibilidade de permitir fazer movimentos para estados com valores mais baixos.
			\item ao ser mais precisa, pois não permite elementos randômicos. 
		\end{enumerate}
	\end{frame}
