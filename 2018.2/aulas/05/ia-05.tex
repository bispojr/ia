	\section{Instrução pelos Colegas}
	
	\begin{frame}{Questão 004}	
		\begin{block}{[Q004]}
			Sobre a definição de problema, qual das alternativas abaixo \underline{não} é um dos componentes da definição?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item estado inicial
			\item ações possíveis
			\item solução ótima
			\item teste de objetivo
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 005}	
		\begin{block}{[Q005]}
			Uma solução para um problema é um caminho desde o estado inicial até um estado objetivo. Chamamos de solução ótima aquela que...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item tem o menor custo de caminho entre todas as soluções.
			\item tem um custo de caminho satisfatório para o usuário.
			\item é um caminho que alcança o estado objetivo.
			\item é um caminho que tem a menor quantidade de nós.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 006}	
		\begin{block}{[Q006]}
			É \underline{incorreto} afirmar sobre a árvore de busca que ela...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item represente todas as possibilidades de caminhos possíveis.
			\item seja igual ao espaço de estados.
			\item possa ter ramos infinitos.
			\item tenha, em cada um de seus nós, a representação de um estado.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 007}	
		\begin{block}{[Q007]}
			Para avaliar o desempenho de um algoritmo de busca específico, qual aspecto não é relevante para se considerar?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item completeza
			\item otimização
			\item complexidade de tempo
			\item descrição do problema
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 008}	
		\begin{block}{[Q008]}
			Em relação à busca cega, qual das alternativas abaixo indica a estratégia de busca que combina a busca em largura com a busca em profundidade?
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item busca de custo uniforme
			\item busca por aprofundamento iterativo
			\item busca em profundidade limitada			
			\item busca bidirecional
		\end{enumerate}
	\end{frame}
