		\begin{block}{[Q029]}
			\begin{columns}
				\begin{column}{0.05\textwidth}\end{column}
				\begin{column}{0.35\textwidth}
					Sobre a descoberta de conhecimento em banco de dados (KDD), é \underline{incorreto} afirmar que...
				\end{column}
				\begin{column}{0.55\textwidth}
					\begin{center}
						\includegraphics[width=1.0\textwidth]{images/kdd}
					\end{center}
				\end{column}
				\begin{column}{0.05\textwidth}\end{column}
			\end{columns}
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item KDD é uma terminologia alternativa para a mineração de dados (MD).
			\item durante a limpeza de dados, efetua-se correções de eventuais inconsistências nos dados.
			\item a etapa de formatação é uma terminologia alternativa para a transformação de dados.
			\item durante a mineração de dados, são extraídas regras e padrões.
		\end{enumerate}
	\end{frame}

		\begin{block}{[Q031]}
			Sobre o tipo de conhecimento gerado pela MD, é correto afirmar que...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item é de natureza dedutiva, inferindo regras e padrões a partir dos dados fornecidos.
			\item é de natureza indutiva, obtendo regras e padrões a partir dos dados fornecidos.
			\item é de natureza dedutiva, pois a partir de proposições gerais obtém-se proposições específicas.
			\item é de natureza indutiva, pois a partir de proposições gerais obtém-se proposições específicas.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 032}	
		\begin{block}{[Q032]}
			A diferença entre as regras de associação e os padrões sequenciais é que...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item as regras de associação detectam associação entre eventos.
			\item os padrões sequenciais detectam associação entre eventos.
			\item os padrões sequenciais detectam associações entre eventos com certos relacionamentos temporais.
			\item as regras de associação detectam associações entre eventos com certos relacionamentos temporais.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 033}	
		\begin{block}{[Q033]}
			Sobre as regras de associação, é \underline{incorreto} afirmar que...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item o suporte refere-se à frequência com que um itemset específico ocorre no banco de dados.
			\item a confiança refere-se à probabilidade de que a regra seja executada, uma vez que o LHS ocorra.
			\item é desejável que os itemsets tenham um suporte grande.
			\item a confiança está mais relacionada a fatores subjetivos do que objetivos em relação ao BD.
		\end{enumerate}
	\end{frame}
