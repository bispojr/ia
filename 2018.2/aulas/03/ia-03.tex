	\section{Instrução pelos Colegas}
	
	\begin{frame}{Questão 001}	
		\begin{block}{[Q001]}
			Sobre agentes e ambientes, é \underline{incorreto} dizer que...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item os agentes agem sobre o ambiente por meio de atuadores.
			\item os agentes percebem o ambiente por meio de sensores.
			\item a percepção refere-se às entradas perceptivas do agente em um dado instante.
			\item a ação de um agente depende das percepções recebidas e não recebidas dos sensores.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 002}	
		\begin{block}{[Q002]}
			Sobre a função de agente, é \underline{correto} afirmar que...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item é uma implementação concreta do comportamento do agente.
			\item ela mapeia qualquer sequência de percepções específica para uma ação.
			\item ela recebe como entrada uma percepção momentânea vinda do ambiente.
			\item ela retorna como saída uma sequência de ações que serão executadas pelo agente.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Questão 003}	
		\begin{block}{[Q003]}
			Sobre a racionalidade, é \underline{incorreto} afirmar que ela depende...
		\end{block}
		\begin{enumerate}[label=(\Alph*)]
			\item da quantidade de sensores que um agente tem à disposição.
			\item das ações que o agente pode executar.
			\item da medida de desempenho que define o critério de sucesso.
			\item da sequência de percepções do agente até o momento.
		\end{enumerate}
	\end{frame}
